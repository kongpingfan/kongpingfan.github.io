struct node {
    short a, b, c;

    node(int a = 0, int b = 0, int c = 0) : a(a), b(b), c(c) {

    }

    bool operator<(const node &other) const {
        if (a != other.a) {
            return a < other.a;
        }
        if (b != other.b) {
            return b < other.b;
        }
        return c < other.c;
    }
};

node que[2][256 * 51 * 51];

class TrySail {
public:

    int get(vector<int> strength) {
        int cur = 0;
        int pre = 1;
        int psz = 1;
        int csz = 0;
        que[pre][psz++] = node(0, 0, 0);
        int len = strength.size();

        for (int i = 0; i < len; i++) {
            int ss = strength[i];
            for (int j = 0; j < psz; j++) {
                que[cur][csz++] = que[pre][j];
                que[cur][csz++] = que[pre][j];
                que[cur][csz++] = que[pre][j];

                que[cur][csz - 1].a ^= ss;
                que[cur][csz - 2].b ^= ss;
                que[cur][csz - 3].c ^= ss;
            }

            sort(que[cur], que[cur] + csz);

            int newsz = 1;
            for (int k = 1; k < csz; k++) {
                if (que[cur][k - 1] < que[cur][k]) {
                    que[cur][newsz++] = que[cur][k];
                }
            }

            csz = newsz;
            swap(cur, pre);
            swap(csz, psz);
            csz = 0;
        }

        int ans = 0;
        for (int i = 0; i < psz; i++) {
            ans = max(ans, que[pre][i].a + que[pre][i].b + que[pre][i].c);
        }
        return ans;
    }
};
